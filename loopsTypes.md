## Loop & Type of Loops

## for , while , do-while , for...of , for...in ##

# Initialization , Condion , Increment/Decrement # 

# for-loop

for(let i = 0; i < 5 ; i ++ ){
    console.log(i);
}

//output : 0 1 2 3 4

# for-of loop

let arr = [1,2,3,4,5];

for(let val of arr){

console.log(val);

}
output: [1,2,3,4,5]
